import { BrowserRouter as Router, Routes, Route } from "react-router-dom";
import DashboardHub from "./pages/DashboardHub";
import EnhancedAdminDashboard from "./pages/EnhancedAdminDashboard";
import PropertyHub from "./pages/PropertyHub";
import FinanceHub from "./pages/FinanceHub";
import StaffSalaries from "./pages/StaffSalaries";

function App() {
  return (
    <Router>
      <Routes>
        {/* Dashboard Hub ‚Üí Default Dashboard */}
        <Route path="/dashboard" element={<DashboardHub />} />

        {/* Enhanced Admin Dashboard ‚Üí Standalone Page */}
        <Route path="/dashboard/admin" element={<EnhancedAdminDashboard />} />

        {/* Other Modules */}
        <Route path="/property" element={<PropertyHub />} />
        <Route path="/finance" element={<FinanceHub />} />
        <Route path="/staff-salaries" element={<StaffSalaries />} />

        {/* Redirect or 404 */}
        <Route path="*" element={<DashboardHub />} />
      </Routes>
    </Router>
  );
}

export default App;
